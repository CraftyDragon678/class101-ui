# TagInput

입력한 텍스트를 태그 형태로 보여주는 인풋입니다.

## Props

<PropsTable of={TagInput} />

## Basic Usage

<Playground>
  {() => {
    class Example extends React.Component {
      
      constructor(props) {
        super(props);
        this.state = {
          value: []
        };
      }
      handleChange(newValue) {
        this.setState({ value: newValue })
      }
      handleSubmit(e) {
        console.log(e)
        e.preventDefault();
      }
      render() {
        return (
          <TagInput separator="," value={this.state.value} onChange={(value) => this.handleChange(value)} allowMessage="Enter와 ','로 구분합니다."/>
        );
      }
    }
    return <Example />
}}

</Playground>
