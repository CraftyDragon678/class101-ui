## onChange

<Playground>
  {() => {
    class Example extends React.Component {
      constructor(props) {
        super(props);
        this.state = {
          checked: false,
        };
      }
      render() {
        return (
          <Checkbox checked={this.state.checked} onChange={e => this.setState({ checked: e.target.checked })}>
            You can check this
          </Checkbox>
        );
      }
    }
    return <Example />;
  }}
</Playground>
